    <script>
        let scene, camera, renderer, humanBody, leftArm, rightArm;
        let isWalking = false, isWaving = false, isTalking = false;
        let clock = new THREE.Clock();
        let synth = window.speechSynthesis;
        let timelineInterval;

        // 🎬 Three.js மூலம் நிஜ 3D மனித உருவத்தை உருவாக்கும் தொழில்நுட்பம்
        function init3DStudio() {
            const container = document.getElementById('canvas3d');
            scene = new THREE.Scene();
            scene.background = new THREE.Color(0xbae6fd); // ஆரம்ப ஸ்கை ப்ளூ பேக்ரவுண்ட்

            camera = new THREE.Camera(); // Orthographic/Perspective கச்சிதமான கேமரா
            camera = new THREE.PerspectiveCamera(45, container.clientWidth / container.clientHeight, 0.1, 1000);
            camera.position.set(0, 1.5, 4);

            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(container.clientWidth, container.clientHeight);
            container.appendChild(renderer.domElement);

            // வெளிச்சம் (Lights)
            const light = new THREE.AmbientLight(0xffffff, 0.8);
            scene.add(light);
            const dirLight = new THREE.DirectionalLight(0xffffff, 0.5);
            dirLight.position.set(5, 10, 5);
            scene.add(dirLight);

            // 👥 மனித உடலமைப்பு உருவாக்கம் (Human Body Model Layers)
            humanBody = new THREE.Group();

            // 1. முகம் (Head)
            const headGeo = new THREE.SphereGeometry(0.3, 32, 32);
            const headMat = new THREE.MeshLambertMaterial({ color: #ffdbac });
            const head = new THREE.Mesh(headGeo, headMat);
            head.position.y = 1.2;
            humanBody.add(head);

            // 2. சட்டை/உடல் (Shirt/Torso)
            const shirtGeo = new THREE.CylinderGeometry(0.3, 0.2, 0.8, 32);
            const shirtMat = new THREE.MeshLambertMaterial({ color: 0xff4757 });
            const shirt = new THREE.Mesh(shirtGeo, shirtMat);
            shirt.position.y = 0.5;
            shirt.name = "shirt";
            humanBody.add(shirt);

            // 3. இடது கை (Left Arm)
            const armGeo = new THREE.CylinderGeometry(0.08, 0.08, 0.6, 16);
            leftArm = new THREE.Mesh(armGeo, headMat);
            leftArm.position.set(-0.4, 0.6, 0);
            humanBody.add(leftArm);

            // 4. வலது கை (Right Arm)
            rightArm = new THREE.Mesh(armGeo, headMat);
            rightArm.position.set(0.4, 0.6, 0);
            humanBody.add(rightArm);

            scene.add(humanBody);
            animate3DLoop();
        }

        // 🔄 3D அனிமேஷன் லூப் இயக்கம் (Real-time Rendering)
        function animate3DLoop() {
            requestAnimationFrame(animate3DLoop);
            let time = clock.getElapsedTime();

            // நடக்கும் அசைவு லாஜிக்
            if (isWalking) {
                humanBody.position.x = Math.sin(time * 2) * 0.5;
                leftArm.rotation.z = Math.sin(time * 5) * 0.5;
                rightArm.rotation.z = -Math.sin(time * 5) * 0.5;
            }
            // கையாட்டும் அசைவு லாஜிக்
            if (isWaving) {
                rightArm.rotation.z = Math.sin(time * 8) * 0.8 + 1;
            }
            // பேசும் அசைவு லாஜிக்
            if (isTalking) {
                humanBody.scale.y = 1 + Math.sin(time * 15) * 0.03;
            }

            renderer.render(scene, camera);
        }

        // 🏃 ஆக்ஷன்களை இயக்குதல்
        function trigger3DAction(act) {
            isWalking = (act === 'walk');
            isWaving = (act === 'wave');
            if (act === 'stop') {
                isWalking = false; isWaving = false;
                leftArm.rotation.set(0,0,0); rightArm.rotation.set(0,0,0);
            }
        }

        // 🎨 முக பாவனைக்கு ஏற்ப சட்டையின் நிறத்தை மாற்றுதல்
        function change3DColor(hexColor) {
            humanBody.children.forEach(child => {
                if(child.name === "shirt") child.material.color.setHex(hexColor);
            });
        }

        // 🔎 கேரக்டரைச் சின்னது/பெருசு மற்றும் நகர்த்தும் வசதி
        function adjust3DCharacter() {
            const scale = document.getElementById('scaleSlider').value;
            const posX = document.getElementById('posSlider').value;
            humanBody.scale.set(scale, scale, scale);
            humanBody.position.x = posX;
        }

        // 🏡 பின்னணி புகைப்படத்தை மாற்றுதல்
        function changeStudioBg(imgUrl) {
            document.getElementById('screenBox').style.background = `url('${imgUrl}') center/cover`;
            document.getElementById('canvas3d').style.opacity = "0.9";
        }

        // 🎙️ வீடியோ எடிட்டிங் டைம்லைன் மற்றும் பேச்சு ஒத்திசைவு
        function speakAndAnimate3D() {
            synth.cancel();
            clearInterval(timelineInterval);
            const text = document.getElementById('speechText').value;
            if(!text) return;

            let utterance = new SpeechSynthesisUtterance(text);
            utterance.lang = 'ta-IN';

            let progress = 0;
            utterance.onstart = () => {
                isTalking = true;
                timelineInterval = setInterval(() => {
                    if (progress < 100) {
                        progress += 5;
                        document.getElementById('tVideo').style.width = progress + '%';
                        document.getElementById('tAudio').style.width = progress + '%';
                    }
                }, 100);
            };

            utterance.onend = () => {
                isTalking = false;
                humanBody.scale.set(1,1,1);
                clearInterval(timelineInterval);
                document.getElementById('tVideo').style.width = '0%';
                document.getElementById('tAudio').style.width = '0%';
            };

            synth.speak(utterance);
        }

        // 20 கேரக்டர்களின் தேர்வுப் பட்டியலை உருவாக்குதல்
        function load20Characters() {
            const list = document.getElementById('charList');
            for(let i=1; i<=20; i++) {
                let card = document.createElement('div');
                card.className = "asset-card";
                card.innerHTML = `👤 மனிதன் மாடல் ${i}`;
                card.onclick = () => {
                    document.querySelectorAll('.asset-card').forEach(c => c.classList.remove('active'));
                    card.classList.add('active');
                    // இங்கு கேரக்டர் மாறும்போது 3D மாடலின் நிறத்தை ரேண்டமாக மாற்றலாம்
                    change3DColor(Math.random() * 0xffffff);
                };
                list.appendChild(card);
            }
        }

        // துவக்கம்
        window.onload = () => {
            init3DStudio();
            load20Characters();
        };
    </script>
